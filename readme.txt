Run: npm install

Create folder "tempFiles"

Create folder "files"

Create file ".env" who contains:
    PORT= (optional, default 8080)
    DB_HOST=
    DB_PORT= (optional, default 3306)
    DB_DATABASE=
    DB_USER=
    DB_PASSWORD= (optionnal, default "")

Run: "node index.js" or "npm run watch" (nodemon)
